26 load:
27  push ebp
28  mov ebp, esp
29 
30  ; ;;;;;;;;;;
31  ;* READ DISK
32  mov bx, [ebp+24]
33  mov es, bx ; segment
34  mov bx, 0x100 ; offset
35  mov dh, 0 ; head
36  mov dl, 0 ; drive
37  mov ch, [ebp+12] ; track
38  mov cl, [ebp+8] ; sector (counted from 1)
39  mov ah, 2 ; BIOS read
40  mov al, [ebp+20] ; sectors to read
41  int 0x13 ; BIOS disk
42  jc .error ; CF = 1 disk reading error
43  cmp al, [ebp+20] ; AL = number of readed sectors
44  jne .error
45 
46  ; ;;;;;;;;;;;;;;;;;;;;
47  ;* PREPARE NEW SEGMENT
48  ; set automatic return from code that didn't return on itself
49  ; put it on first non program position - (size*512)+0x100+4
50  mov ax, [ebp+20]
51  imul ax, 512
52  add ax, 0x104
53  mov bx, ax
54  mov DWORD es:[bx], 0x900008b8 ;mov ax,0x8 - ENOEXEC
55  mov DWORD es:[bx+4],0x900002c2 ;ret 2
56 
57  ; load at 0x2000:80 program parameters
58  mov si, WORD[ebp+16] ; pointer to parameter string in si
59  xor bx, bx
60  .loop:
61  mov al, [si+bx]
62  mov es:[bx+0x80],al
63  inc bx
64  cmp bx, 0x7f
65  jge .after
66  cmp al, 0
67  jne .loop
68  .after:
69  mov BYTE es:[0xff], 0 ;set null character
70 
71  ; ;;;;;;;;;;;;;;;;;
72  ; RUN NEW SEGMENT
73  setSegments es
74  ; set COM header like in https://en.wikipedia.org/wiki/Program_Segment_Prefix
75  ;00000000 9090 nop nop
76  ;00000002 6A00 push byte 0x0 ; because C uses 32 bit return, assembly must return ret 2 to pop this
77  ;00000004 E8F900 call 0x100
78  ;00000007 CB retf
79  mov DWORD[0x0], 0x006a9090
80  mov DWORD[0x4], 0xcb00f9e8
81  mov DWORD[0x8], .end
82  mov DWORD[0x12],ebp
83 
84  mov WORD[0x16], 0 ; CALL OFFSET
85  mov WORD[0x18], es ; CALL SEGMENT
86  mov esp, 0xffff ; set stack begin on end of segment
87  mov ebp, esp
88  pushf
89  call far [es:0x16] ; push cs; push ip
90  popf
91  .end:
92  ; in caller CS
93  mov dx, ax ;save return code
94  xchg esp, [0x12]
95  setSegments cs
96 
97  pop ebp
98  mov ax, dx
99  ret
100 
101  .error:
102  ; return -ENOEXEC
103  pop ebp
104  mov ax, - 8 ; ENOEXEC - executable format error
105  ret
