3 ; @brief Kernel's interruptions
4 ; @details About interrupts:
5 ; @details When interrupt handler is called on stack are: [esp]=old eip [esp+2]=old cs [esp+4] old flags
6 ; @details to use local interrupt's variables DS must be changed to local i.e. KERNEL_ADDRESS
7 ; @details all cs of interrupt handler are KERNEL_ADDRESS - only kernel should handle

16 ; @brief Set system interruptions
17 ; @details set syscall and errors handles
18 ;;
19 setInterrupts:
20  push ebp
21  mov ebp, esp
22  push ds
23 
24  ;handle 0 division
25  mov bx, 0
26  mov ds, bx
27  mov WORD[ds:bx+2], KERNEL_ADDRESS
28  mov WORD[ds:bx], divZero
29  ;handle syscall
30  mov bx, 0x20*4
31  mov WORD[ds:bx+2], KERNEL_ADDRESS
32  mov WORD[ds:bx], syscall
33 
34  pop ds
35  mov esp, ebp
36  pop ebp
37  ret

39 ;;
40 ; @brief add interruption
41 ; @details set *(number*4)=function
42 ; @details set *(number*4+2)=KERNEL_ADDRESS - only kernel can handle interruptions
43 ; @param number new interruption number [ebp+8]
44 ; @param function interruption handler [ebp+12]
45 ;;
46 addInterrupt:
47  push ebp
48  mov ebp, esp
49  push ds
50 
51  mov bx, 0
52  mov ds, bx
53  mov ax, [ebp+8] ; set ax as parameter then ax*=4
54  mov bx, 4
55  mul bx
56  mov bx, ax
57  mov ax, WORD[ebp+12]
58  mov WORD[ds:bx], ax
59  mov WORD[ds:bx+2], KERNEL_ADDRESS
60 
61  pop ds
62  mov esp, ebp
63  pop ebp
64  ret
65 
66 ;;
67 ; @brief System interruption 0x20
68 ; @details it adds:
69 ; @details 0 - putc AL = character
70 ; @details 1 - puts BX = &string
71 ; @details 2 - puti BX = unsigned number
72 ; @details 3 - printf BX = pointer to first parameter in stack, nth parameter is on BX+4n
73 ;;
74 syscall:
75  enter 0, 0
76  ; system calls
77  cmp ah, 0
78  je .putc
79  cmp ah, 1
80  je .puts
81  cmp ah, 2
82  je .puti
83  cmp ah, 3
84  je .printf
85  ; if not recognised print message
86  setDS KERNEL_ADDRESS
87  push .message
88  call puts
89  add sp, 2
90  pop ds
91  jmp .exit
92  .message db "SYSCALL YAY!",0xa,0

97  push ax
98  call putc
99  add sp, 2
100  jmp .exit
101 
102  .puts:
103  ; bx is address of string to put
104  push bx
105  call puts
106  add sp, 2
107  jmp .exit
108 
109  .puti:
110  ; bx is number to print
111  push bx
112  call puti
113  add sp, 2
114  jmp .exit
115 
116  .printf:
117  push WORD 0
118  push bx

122  leave

125 divZero:
126  ; zero division handler
127  dbg
128  setDS KERNEL_ADDRESS
129  push .err
130  call puts
131  add sp, 2
132  pop ds
133 
134  cmp esi, 0 ; when C divides it uses idiv - 3 byte and divides by esi
135  jne .Csrc

137  .Csrc: ; if it is 'normal' div add 2 bytes

139
